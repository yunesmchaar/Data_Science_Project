# src/security/anonymizer.py
import hashlib, secrets
import pandas as pd

class MedicalAnonymizer:
    def __init__(self, k=5):
        self.k = k
        self.salt = secrets.token_hex(32)

    def pseudonymize_id(self, value):
        """SHA-256 + sel aléatoire"""
        return 'PT-' + hashlib.sha256(
            (self.salt + str(value)).encode()).hexdigest()[:8]

    def verify_k_anonymity(self, df, quasi_ids):
        groups = df.groupby(quasi_ids).size()
        return bool((groups >= self.k).all())
